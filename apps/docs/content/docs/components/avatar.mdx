## Usage

```tsx
import { Avatar, AvatarFallback, AvatarImage } from "@bigcomponents/core";
```

```tsx
<Avatar>
  <AvatarImage src="https://github.com/Gabriel200622.png" alt="@biggabo" />
  <AvatarFallback>BG</AvatarFallback>
</Avatar>
```
